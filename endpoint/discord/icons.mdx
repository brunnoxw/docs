title: "icons"
api: "GET http://147.rest/icons?tipo=aleatorios&quantidade=10"

---

### Headers

<ParamField header="X-API_key" type="string" required>
	Enviar, em cada request, a sua API KEY no cabeçalho pedido
</ParamField>

### Query
<ParamField query="tipo" type="string" placeholder="aleatorios" required>
	Categoria de icones
</ParamField>
<ParamField query="quantidade" type="number" placeholder="10" required>
	Quantidade de icones que vai retornar
</ParamField>
### Status

<ResponseField name="200" type="number">
	Retorna `200` se tudo der certo.
</ResponseField>
<ResponseField name="400" type="number">
	Retorna `400` para Parâmetro faltando;
</ResponseField>
<ResponseField name="422" type="number">
	Retorna `422` para parametro inválido.
</ResponseField>

<ResponseField name="sucesso" type="boolean">
	Retorna `true` para sucesso e `false` para erro.
</ResponseField>
<ResponseField name="erro" type="string">
	Retorna a mensagem do erro
</ResponseField>
